Export Results

Exporting Performance Test Results

Overview

In this section, we’ll implement mechanisms to export and store performance test results. We’ll create a comprehensive data collection system that captures test metrics, generates reports, and stores historical data for trend analysis.

Implementation Steps

1. Configure Result Export

  1. Create result export script:
// export-results.js
const fs = require('fs');
const AWS = require('aws-sdk');
const cloudwatch = new AWS.CloudWatch();
const s3 = new AWS.S3();

async function exportResults(testResults, options) {
  const timestamp = new Date().toISOString();
  const metrics = [];

  // Format metrics for CloudWatch
  Object.entries(testResults.metrics).forEach(([key, value]) => {
    metrics.push({
      MetricName: key,
      Value: value,
      Unit: 'None',
      Timestamp: timestamp,
      Dimensions: [
        {
          Name: 'TestType',
          Value: options.testType
        }
      ]
    });
  });

  // Send metrics to CloudWatch
  await cloudwatch.putMetricData({
    Namespace: 'PerformanceTests',
    MetricData: metrics
  }).promise();

  // Save detailed results to S3
  await s3.putObject({
    Bucket: options.resultsBucket,
    Key: `${options.testType}/${timestamp}.json`,
    Body: JSON.stringify(testResults),
    ContentType: 'application/json'
  }).promise();
}

module.exports = { exportResults };

2. Create Results Database

  1. Create DynamoDB table:
aws dynamodb create-table \
  --table-name PerformanceTestResults \
  --attribute-definitions \
    AttributeName=TestId,AttributeType=S \
    AttributeName=Timestamp,AttributeType=S \
  --key-schema \
    AttributeName=TestId,KeyType=HASH \
    AttributeName=Timestamp,KeyType=RANGE \
  --provisioned-throughput ReadCapacityUnits=5,WriteCapacityUnits=5
  1. Create database storage script:
// store-results.js
const AWS = require('aws-sdk');
const dynamodb = new AWS.DynamoDB.DocumentClient();

async function storeResults(testResults) {
  const params = {
    TableName: 'PerformanceTestResults',
    Item: {
      TestId: testResults.testId,
      Timestamp: new Date().toISOString(),
      TestType: testResults.testType,
      Duration: testResults.duration,
      Metrics: {
        ResponseTime: testResults.metrics.http_req_duration,
        ErrorRate: testResults.metrics.http_req_failed,
        Throughput: testResults.metrics.http_reqs
      },
      Environment: testResults.env,
      Status: testResults.status
    }
  };

  await dynamodb.put(params).promise();
}

const Handlebars = require('handlebars');
const Chart = require('chart.js');

async function generateReport(testResults, templatePath, outputPath) {
  // Read template
  const template = Handlebars.compile(
    fs.readFileSync(templatePath, 'utf-8')
  );

  // Generate charts
  const charts = await generateCharts(testResults);

  // Create report data
  const reportData = {

  // Generate HTML

}

Best Practices

  1. Data Organization

    • Use consistent naming
    • Implement versioning
    • Structure data hierarchically
    • Document data format

  2. Storage Management

    • Implement retention policies
    • Monitor storage usage
    • Archive old data
    • Backup important results

  3. Report Generation

    • Include key metrics
    • Add visualizations
    • Highlight anomalies
    • Provide context
